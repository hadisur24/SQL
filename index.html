🏗️ Database & Table Basics

Learn to create databases, tables, and understand data types.

📄 database-setup.sql 
-- Create a new database
CREATE DATABASE company_db;
USE company_db;

-- Create employees table with proper data types
CREATE TABLE employees (
    emp_id       INT PRIMARY KEY AUTO_INCREMENT,
    first_name   VARCHAR(50) NOT NULL,
    last_name    VARCHAR(50) NOT NULL,
    email        VARCHAR(100) UNIQUE,
    department   VARCHAR(50),
    salary       DECIMAL(10,2),
    hire_date    DATE,
    manager_id   INT,
    is_active    BOOLEAN DEFAULT TRUE
);

-- Create departments table
CREATE TABLE departments (
    dept_id      INT PRIMARY KEY AUTO_INCREMENT,
    dept_name    VARCHAR(50) NOT NULL,
    location     VARCHAR(100),
    budget       DECIMAL(15,2)
);
💡
Pro Tip: Common Data Types

INT → Numbers | VARCHAR(n) → Text | DECIMAL(p,s) → Money | DATE → Dates | BOOLEAN → True/False | TEXT → Long text

🔍 WHERE Clause & Filtering

📄 where-filtering.sql 
-- Comparison operators
SELECT * FROM employees WHERE salary > 70000;
SELECT * FROM employees WHERE salary >= 70000;
SELECT * FROM employees WHERE salary <> 65000;  -- Not equal

-- AND, OR, NOT
SELECT * FROM employees
WHERE department = 'IT' AND salary > 60000;

SELECT * FROM employees
WHERE department = 'IT' OR department = 'HR';

SELECT * FROM employees
WHERE NOT department = 'Sales';

-- IN operator (multiple values)
SELECT * FROM employees
WHERE department IN ('IT', 'HR', 'Finance');

-- BETWEEN (range - inclusive)
SELECT * FROM employees
WHERE salary BETWEEN 50000 AND 80000;

-- LIKE (pattern matching)
SELECT * FROM employees WHERE first_name LIKE 'R%';     -- Starts with R
SELECT * FROM employees WHERE last_name LIKE '%ar%';   -- Contains 'ar'
SELECT * FROM employees WHERE first_name LIKE '_____';  -- Exactly 5 chars
SELECT * FROM employees WHERE email LIKE '%@gmail.com';

-- IS NULL / IS NOT NULL
SELECT * FROM employees WHERE manager_id IS NULL;
SELECT * FROM employees WHERE manager_id IS NOT NULL;

-- ORDER BY (sorting)
SELECT * FROM employees ORDER BY salary DESC;
SELECT * FROM employees ORDER BY department ASC, salary DESC;

-- LIMIT & OFFSET
SELECT * FROM employees ORDER BY salary DESC LIMIT 5;
SELECT * FROM employees LIMIT 10 OFFSET 5; -- Skip first 5

💻 JOIN Code Examples

📄 all-joins.sql 
-- ✅ INNER JOIN: Employees with their department names
SELECT
    e.first_name,
    e.salary,
    d.dept_name,
    d.location
FROM employees e
INNER JOIN departments d ON e.dept_id = d.dept_id;

-- ✅ LEFT JOIN: ALL employees (even those without a department)
SELECT
    e.first_name,
    COALESCE(d.dept_name, 'Unassigned') AS department
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.dept_id;

-- ✅ SELF JOIN: Find each employee's manager
SELECT
    e.first_name AS employee,
    e.salary AS emp_salary,
    m.first_name AS manager,
    m.salary AS mgr_salary
FROM employees e
LEFT JOIN employees m ON e.manager_id = m.emp_id;

-- ✅ Multiple JOINs: Employee + Department + Projects
SELECT
    e.first_name,
    d.dept_name,
    p.project_name,
    p.deadline
FROM employees e
JOIN departments d ON e.dept_id = d.dept_id
JOIN projects p ON e.emp_id = p.lead_id
WHERE p.status = 'Active'
ORDER BY p.deadline;

📊 Aggregate Functions & GROUP BY

📄 aggregations.sql 
-- Basic aggregate functions
SELECT
    COUNT(*) AS total_employees,
    AVG(salary) AS avg_salary,
    SUM(salary) AS total_payroll,
    MIN(salary) AS lowest_salary,
    MAX(salary) AS highest_salary
FROM employees;

-- GROUP BY: Aggregate per department
SELECT
    department,
    COUNT(*) AS emp_count,
    ROUND(AVG(salary), 2) AS avg_salary,
    MAX(salary) AS top_salary
FROM employees
GROUP BY department
ORDER BY avg_salary DESC;

-- HAVING: Filter AFTER aggregation
SELECT
    department,
    COUNT(*) AS emp_count,
    AVG(salary) AS avg_salary
FROM employees
WHERE hire_date > '2022-01-01'     -- filter rows FIRST
GROUP BY department
HAVING COUNT(*) > 3               -- filter groups AFTER
ORDER BY avg_salary DESC;
⚠️
WHERE vs HAVING — Common Interview Question!

WHERE filters individual rows BEFORE grouping. HAVING filters groups AFTER aggregation. You cannot use aggregate functions in WHERE.

📙 PHASE 2B — WEEK 3

Subqueries & CASE WHEN

Write nested queries and conditional logic to solve complex problems

🔄 Subquery Types

📌 Subquery in WHERE

  • Filter based on another query's result
  • Use with =, IN, EXISTS
  • Most common type

📌 Subquery in FROM

  • Creates a "derived table"
  • Must have an alias
  • Used for complex aggregations

📌 Subquery in SELECT

  • Returns a single value per row
  • Good for comparisons
  • Called scalar subquery

📌 Correlated Subquery

  • References the outer query
  • Runs once per outer row
  • Slower but powerful
📄 subqueries.sql 
-- ✅ Subquery in WHERE: Employees earning above average
SELECT first_name, salary
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);

-- ✅ Subquery with IN
SELECT first_name, department
FROM employees
WHERE dept_id IN (
    SELECT dept_id FROM departments WHERE location = 'Mumbai'
);

-- ✅ Correlated Subquery: Employees earning more than dept average
SELECT first_name, salary, department
FROM employees e1
WHERE salary > (
    SELECT AVG(salary)
    FROM employees e2
    WHERE e2.department = e1.department
);

-- ✅ EXISTS: Departments that have employees
SELECT dept_name
FROM departments d
WHERE EXISTS (
    SELECT 1 FROM employees e
    WHERE e.dept_id = d.dept_id
);

🔀 CASE WHEN — Conditional Logic

📄 case-when.sql 
-- ✅ Categorize employees by salary
SELECT
    first_name,
    salary,
    CASE
        WHEN salary >= 100000 THEN '💎 Premium'
        WHEN salary >= 70000  THEN '🥇 Senior'
        WHEN salary >= 45000  THEN '🥈 Mid-Level'
        ELSE '🥉 Junior'
    END AS salary_tier
FROM employees
ORDER BY salary DESC;

-- ✅ CASE WHEN with aggregation (Pivot-style)
SELECT
    department,
    COUNT(CASE WHEN salary >= 70000 THEN 1 END) AS high_earners,
    COUNT(CASE WHEN salary < 70000 THEN 1 END) AS regular,
    COUNT(*) AS total
FROM employees
GROUP BY department;
📕 PHASE 3 — WEEK 4-6

Window Functions & CTEs

The game-changing skill that separates beginners from advanced SQL users

🎯
Why Window Functions Matter

Window functions perform calculations across rows WITHOUT collapsing them (unlike GROUP BY). They're asked in 80%+ of data analyst SQL interviews!

📊 Window Functions Comparison

Function Purpose Handles Ties Example Output
ROW_NUMBER() Unique sequential number No ties (always unique) 1, 2, 3, 4, 5
RANK() Rank with gaps Same rank, skips next 1, 2, 2, 4, 5
DENSE_RANK() Rank without gaps Same rank, no skip 1, 2, 2, 3, 4
NTILE(n) Divide into n buckets Distributes evenly 1, 1, 2, 2, 3, 3
LAG(col, n) Previous row's value N/A NULL, 100, 200, 150
LEAD(col, n) Next row's value N/A 200, 150, 300, NULL

💻 Window Functions in Action

📄 window-functions.sql 
-- ═══════════════════════════════════════════════════
-- SYNTAX: FUNCTION() OVER (PARTITION BY x ORDER BY y)
-- ═══════════════════════════════════════════════════

-- ✅ Ranking employees by salary within each department
SELECT
    first_name,
    department,
    salary,
    ROW_NUMBER() OVER (
        PARTITION BY department
        ORDER BY salary DESC
    ) AS dept_rank,
    DENSE_RANK() OVER (
        ORDER BY salary DESC
    ) AS overall_rank
FROM employees;

-- ✅ LAG & LEAD: Compare with previous/next row
SELECT
    first_name,
    hire_date,
    salary,
    LAG(salary, 1) OVER (ORDER BY hire_date) AS prev_salary,
    LEAD(salary, 1) OVER (ORDER BY hire_date) AS next_salary,
    salary - LAG(salary, 1) OVER (ORDER BY hire_date) AS change
FROM employees;

-- ✅ Running Total & Running Average
SELECT
    order_date,
    revenue,
    SUM(revenue) OVER (
        ORDER BY order_date
    ) AS running_total,
    AVG(revenue) OVER (
        ORDER BY order_date
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS moving_avg_3day
FROM daily_sales;

📦 Common Table Expressions (CTEs)

CTEs make complex queries readable and reusable.

📄 ctes.sql 
-- ✅ CTE + Window Function: Top 3 earners per department
WITH ranked_employees AS (
    SELECT
        first_name,
        department,
        salary,
        DENSE_RANK() OVER (
            PARTITION BY department
            ORDER BY salary DESC
        ) AS dept_rank
    FROM employees
)
SELECT * FROM ranked_employees
WHERE dept_rank <= 3;

-- ✅ Multiple CTEs: Complex analysis
WITH dept_stats AS (
    SELECT
        department,
        AVG(salary) AS avg_salary,
        COUNT(*) AS emp_count
    FROM employees
    GROUP BY department
),
company_avg AS (
    SELECT AVG(salary) AS overall_avg
    FROM employees
)
SELECT
    ds.department,
    ds.emp_count,
    ROUND(ds.avg_salary, 2) AS dept_avg,
    ROUND(ca.overall_avg, 2) AS company_avg,
    CASE
        WHEN ds.avg_salary > ca.overall_avg THEN '✅ Above Average'
        ELSE '❌ Below Average'
    END AS status
FROM dept_stats ds
CROSS JOIN company_avg ca
ORDER BY ds.avg_salary DESC;

Real Business Patterns

Master the query patterns that companies actually use in production

📈 Pattern 1: Year-over-Year Growth Analysis ▼
📄 yoy-growth.sql 
WITH yearly_revenue AS (
    SELECT
        YEAR(order_date) AS year,
        SUM(amount) AS total_revenue
    FROM orders
    GROUP BY YEAR(order_date)
)
SELECT
    year,
    total_revenue,
    LAG(total_revenue) OVER (ORDER BY year) AS prev_year,
    ROUND(
        (total_revenue - LAG(total_revenue) OVER (ORDER BY year))
        * 100.0
        / LAG(total_revenue) OVER (ORDER BY year), 2
    ) AS yoy_growth_pct
FROM yearly_revenue;
🔄 Pattern 2: Customer Churn Analysis ▼
📄 churn-analysis.sql 
-- Find customers who purchased in Jan but NOT in Feb (churned)
SELECT customer_id
FROM orders
WHERE DATE_FORMAT(order_date, '%Y-%m') = '2024-01'
AND customer_id NOT IN (
    SELECT customer_id
    FROM orders
    WHERE DATE_FORMAT(order_date, '%Y-%m') = '2024-02'
);

-- Churn rate calculation
WITH jan_customers AS (
    SELECT DISTINCT customer_id
    FROM orders
    WHERE DATE_FORMAT(order_date, '%Y-%m') = '2024-01'
),
feb_customers AS (
    SELECT DISTINCT customer_id
    FROM orders
    WHERE DATE_FORMAT(order_date, '%Y-%m') = '2024-02'
)
SELECT
    COUNT(j.customer_id) AS jan_total,
    COUNT(f.customer_id) AS retained,
    COUNT(j.customer_id) - COUNT(f.customer_id) AS churned,
    ROUND(
        (COUNT(j.customer_id) - COUNT(f.customer_id)) * 100.0
        / COUNT(j.customer_id), 2
    ) AS churn_rate_pct
FROM jan_customers j
LEFT JOIN feb_customers f
    ON j.customer_id = f.customer_id;
🔍 Pattern 3: Find Duplicates & Remove Them ▼
📄 find-duplicates.sql 
-- Method 1: GROUP BY to find duplicates
SELECT email, COUNT(*) AS cnt
FROM employees
GROUP BY email
HAVING COUNT(*) > 1;

-- Method 2: Window Function to find & remove duplicates
WITH duplicates AS (
    SELECT *,
        ROW_NUMBER() OVER (
            PARTITION BY email
            ORDER BY emp_id
        ) AS rn
    FROM employees
)
SELECT * FROM duplicates WHERE rn > 1;  -- these are duplicates
📊 Pattern 4: Pivot Table in SQL ▼
📄 pivot-table.sql 
-- Pivot: Monthly revenue by product
SELECT
    product_name,
    SUM(CASE WHEN MONTH(order_date) = 1 THEN amount ELSE 0 END) AS Jan,
    SUM(CASE WHEN MONTH(order_date) = 2 THEN amount ELSE 0 END) AS Feb,
    SUM(CASE WHEN MONTH(order_date) = 3 THEN amount ELSE 0 END) AS Mar,
    SUM(amount) AS Total
FROM orders
WHERE YEAR(order_date) = 2024
GROUP BY product_name
ORDER BY Total DESC;
🏆 Pattern 5: Top N per Group ▼
📄 top-n-per-group.sql 
-- Top 3 highest paid employees per department
WITH ranked AS (
    SELECT
        first_name,
        department,
        salary,
        DENSE_RANK() OVER (
            PARTITION BY department
            ORDER BY salary DESC
        ) AS rnk
    FROM employees
)
SELECT * FROM ranked WHERE rnk <= 3;
⚡ PHASE 5 — WEEK 9

Query Optimization

Write faster, more efficient SQL queries

❌ Bad vs ✅ Good Practices

📄 optimization-tips.sql 
-- ❌ BAD: SELECT * (fetches all columns)
SELECT * FROM employees;

-- ✅ GOOD: Select only needed columns
SELECT first_name, salary FROM employees;

-- ─────────────────────────────────────────────

-- ❌ BAD: Function on indexed column in WHERE
SELECT * FROM employees
WHERE YEAR(hire_date) = 2024;

-- ✅ GOOD: Use range comparison instead
SELECT * FROM employees
WHERE hire_date >= '2024-01-01'
  AND hire_date < '2025-01-01';

-- ─────────────────────────────────────────────

-- ❌ BAD: Using OR for multiple values
SELECT * FROM employees
WHERE dept_id = 1 OR dept_id = 2 OR dept_id = 3;

-- ✅ GOOD: Use IN operator
SELECT * FROM employees
WHERE dept_id IN (1, 2, 3);

-- ─────────────────────────────────────────────

-- ❌ BAD: LIKE with leading wildcard (can't use index)
SELECT * FROM employees WHERE name LIKE '%kumar';

-- ✅ GOOD: LIKE with trailing wildcard (can use index)
SELECT * FROM employees WHERE name LIKE 'Kumar%';

-- ─────────────────────────────────────────────

-- 💡 Create indexes for frequently filtered columns
CREATE INDEX idx_department ON employees(department);
CREATE INDEX idx_salary ON employees(salary);
CREATE INDEX idx_dept_salary ON employees(department, salary);

-- 💡 Check query execution plan
EXPLAIN ANALYZE
SELECT * FROM employees WHERE department = 'IT';

Top Interview Questions

Must-know SQL problems asked by top companies

🏅 Classic Interview Problems

Q1: Find the Second Highest Salary 3 Methods ▼
📄 second-highest-salary.sql 
-- Method 1: Subquery
SELECT MAX(salary) AS second_highest
FROM employees
WHERE salary < (SELECT MAX(salary) FROM employees);

-- Method 2: LIMIT OFFSET
SELECT DISTINCT salary
FROM employees
ORDER BY salary DESC
LIMIT 1 OFFSET 1;

-- Method 3: DENSE_RANK (Best for Nth highest)
WITH ranked AS (
    SELECT salary,
        DENSE_RANK() OVER (ORDER BY salary DESC) AS rnk
    FROM employees
)
SELECT DISTINCT salary FROM ranked WHERE rnk = 2;
Q2: Employees Earning More Than Their Manager Self JOIN ▼
📄 emp-vs-manager.sql 
SELECT
    e.first_name AS employee,
    e.salary AS emp_salary,
    m.first_name AS manager,
    m.salary AS mgr_salary
FROM employees e
JOIN employees m ON e.manager_id = m.emp_id
WHERE e.salary > m.salary;
Q3: Customers Who Never Ordered LEFT JOIN ▼
📄 never-ordered.sql 
-- Method 1: LEFT JOIN + IS NULL
SELECT c.customer_name
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
WHERE o.order_id IS NULL;

-- Method 2: NOT EXISTS
SELECT customer_name
FROM customers c
WHERE NOT EXISTS (
    SELECT 1 FROM orders o
    WHERE o.customer_id = c.customer_id
);
Q4: Department Top 3 Salaries Window + CTE ▼
📄 dept-top-3.sql 
WITH ranked AS (
    SELECT
        d.dept_name AS Department,
        e.first_name AS Employee,
        e.salary AS Salary,
        DENSE_RANK() OVER (
            PARTITION BY d.dept_name
            ORDER BY e.salary DESC
        ) AS rnk
    FROM employees e
    JOIN departments d ON e.dept_id = d.dept_id
)
SELECT Department, Employee, Salary
FROM ranked
WHERE rnk <= 3;
